Inductive list (T : Type) : Type :=
| nil : list T
| cons :
    T $\rightarrow$ list T $\rightarrow$ list T.

Inductive vector (T : Type) : (@\codediff{nat}@) -> Type :=
| nil : vector T (@\codediff{O}@)
| cons :
    T $\rightarrow$ $\forall$ (@\codediff{(n : nat)}@), vector T (@\codediff{n}@) $\rightarrow$ vector T (@\codediff{(S n)}@).
